-- This is an example shotgun weapon that inherits from the ax_base weapon.
-- It demonstrates advanced features like individual shell reloading and pump action.
-- Shotguns fire multiple pellets per shot for close-range damage.
-- To create your own shotgun, copy this file and modify the values below.

AddCSLuaFile()

-- Basic weapon information that appears in the spawn menu
SWEP.Base = "ax_base"  -- This tells the weapon to inherit from our enhanced base weapon
SWEP.PrintName = "Parallax Shotgun"  -- The name that shows up in the spawn menu
SWEP.Author = "Riggs"  -- Your name or team name
SWEP.Category = "Parallax"  -- The category in the spawn menu

-- Primary fire settings - this controls how the weapon shoots
SWEP.Primary = {
    ClipSize = 6,         -- How many shells fit in the weapon (smaller than other weapons)
    DefaultClip = 12,     -- How many shells the player starts with (2 reloads worth)
    Automatic = false,    -- false = semi-automatic (one shot per click)
    Ammo = "Buckshot",    -- The type of ammo this weapon uses (shotgun shells)
    Delay = 0.8,          -- Time between shots in seconds (slower firing rate)
    Damage = 8,           -- Damage per pellet (multiplied by NumShots for total damage)
    Cone = 0.1,           -- How accurate the weapon is (very spread out for shotguns)
    Recoil = 2.0,         -- How much the weapon kicks back (high recoil)
    Sound = Sound("Weapon_Shotgun.Single"),  -- The sound played when firing
    SoundEmpty = Sound("Weapon_Shotgun.Empty"),  -- Sound when trying to fire with no ammo
    NumShots = 8,         -- How many pellets fired per shot (creates the spread pattern)
    TracerName = "Tracer" -- The effect for bullet tracers
}

-- Secondary fire settings - usually not used for basic weapons
SWEP.Secondary = {
    ClipSize = -1,        -- -1 means no secondary ammo
    DefaultClip = -1,
    Automatic = false,
    Ammo = "none"         -- No secondary ammo type
}

-- Weapon models and animations
SWEP.ViewModel = "models/weapons/c_shotgun.mdl"   -- The model shown in first person view
SWEP.WorldModel = "models/weapons/w_shotgun.mdl"  -- The model shown in third person/world
SWEP.HoldType = "shotgun"  -- How the player holds the weapon (shotgun-style animations)

-- Iron sights settings - for aiming down sights
SWEP.IronSightsEnabled = true  -- Enable/disable iron sights
SWEP.IronSightsPos = Vector(-8.5, -8, 4.5)  -- Position offset when aiming (different for shotgun)
SWEP.IronSightsAng = Angle(-1.3, 0, 0)      -- Angle offset when aiming
SWEP.IronSightsFOV = 0.8     -- Field of view when aiming (lower = more zoomed)
SWEP.IronSightsSensitivity = 0.5  -- Mouse sensitivity when aiming
SWEP.IronSightsToggle = false     -- false = hold to aim, true = toggle to aim
SWEP.IronSightsDelay = 0.25       -- Time it takes to enter/exit iron sights

-- Special shotgun features from the ax_base weapon
SWEP.ShotgunReload = true  -- Enable individual shell reloading (one shell at a time)
SWEP.PumpAfterShoot = true -- Enable pump action animation/sound after firing

-- Muzzle light color - the flash effect when firing
-- This is a feature from the ax_base weapon
SWEP.MuzzleLightColor = Color(255, 200, 150)  -- Orange-yellow light color
