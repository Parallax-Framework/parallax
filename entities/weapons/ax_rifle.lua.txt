-- This is an example automatic rifle weapon that inherits from the ax_base weapon.
-- It demonstrates how to create a weapon that fires continuously when holding the trigger.
-- To create your own automatic weapon, copy this file and modify the values below.

AddCSLuaFile()

-- Basic weapon information that appears in the spawn menu
SWEP.Base = "ax_base"  -- This tells the weapon to inherit from our enhanced base weapon
SWEP.PrintName = "Parallax Rifle"  -- The name that shows up in the spawn menu
SWEP.Author = "Riggs"  -- Your name or team name
SWEP.Category = "Parallax"  -- The category in the spawn menu
SWEP.Spawnable = true -- Whether players can spawn this weapon
SWEP.AdminOnly = false -- Whether only admins can spawn this weapon

-- Primary fire settings - this controls how the weapon shoots
SWEP.Primary = {
    ClipSize = 30,        -- How many bullets fit in one magazine (larger than pistol)
    DefaultClip = 90,     -- How many bullets the player starts with (3 magazines worth)
    Automatic = true,     -- true = automatic fire (continues firing while holding trigger)
    Ammo = "AR2",         -- The type of ammo this weapon uses (rifle ammo)
    Delay = 0.1,          -- Time between shots in seconds (faster than pistol)
    Damage = 25,          -- How much damage each bullet does (higher than pistol)
    Cone = 0.03,          -- How accurate the weapon is (slightly less accurate than pistol)
    Recoil = 0.5,         -- How much the weapon kicks back when fired (less than pistol)
    Sound = Sound("Weapon_AR2.Single"),  -- The sound played when firing
    SoundEmpty = Sound("Weapon_AR2.Empty"),  -- Sound when trying to fire with no ammo
    NumShots = 1,         -- How many bullets fired per shot (1 = single bullet)
    TracerName = "Tracer" -- The effect for bullet tracers
}

-- Secondary fire settings - usually not used for basic weapons
SWEP.Secondary = {
    ClipSize = -1,        -- -1 means no secondary ammo
    DefaultClip = -1,
    Automatic = false,
    Ammo = "none"         -- No secondary ammo type
}

-- Weapon models and animations
SWEP.ViewModel = "models/weapons/c_irifle.mdl"  -- The model shown in first person view
SWEP.WorldModel = "models/weapons/w_irifle.mdl" -- The model shown in third person/world
SWEP.HoldType = "ar2"   -- How the player holds the weapon (rifle-style animations)

-- Iron sights settings - for aiming down sights
SWEP.IronSightsEnabled = true  -- Enable/disable iron sights
SWEP.IronSightsPos = Vector(-5.8, -8, 3)  -- Position offset when aiming
SWEP.IronSightsAng = Angle(-1.3, 0, 0)    -- Angle offset when aiming
SWEP.IronSightsFOV = 0.8     -- Field of view when aiming (lower = more zoomed)
SWEP.IronSightsSensitivity = 0.5  -- Mouse sensitivity when aiming
SWEP.IronSightsToggle = false     -- false = hold to aim, true = toggle to aim
SWEP.IronSightsDelay = 0.25       -- Time it takes to enter/exit iron sights

-- Muzzle light color - the flash effect when firing
-- This is a feature from the ax_base weapon
SWEP.MuzzleLightColor = Color(255, 200, 150)  -- Orange-yellow light color
